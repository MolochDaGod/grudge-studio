        // ============================================
        // 1. Storage Manager
        // ============================================
        
        class PuterStorageManager {
            constructor() {
                this.appFolder = '/rts-island-game';
                this.initialized = false;
            }

            async init() {
                try {
                    this.initialized = true;
                    console.log('✓ Storage initialized');
                } catch (error) {
                    console.error('Storage init error:', error);
                }
            }

            async saveGameState(gameState) {
                try {
                    const key = `game-state-${Date.now()}`;
                    localStorage.setItem(key, JSON.stringify(gameState));
                    console.log(`✓ Game saved: ${key}`);
                    return { success: true, filename: key };
                } catch (error) {
                    console.error('Save error:', error);
                    return { success: false, error: error.message };
                }
            }

            async loadLatestGameState() {
                try {
                    const keys = Object.keys(localStorage).filter(k => k.startsWith('game-state-'));
                    if (keys.length === 0) return null;
                    
                    const latestKey = keys.sort().reverse()[0];
                    const data = localStorage.getItem(latestKey);
                    return JSON.parse(data);
                } catch (error) {
                    console.error('Load error:', error);
                    return null;
                }
            }

            async listSavedGames() {
                try {
                    const keys = Object.keys(localStorage).filter(k => k.startsWith('game-state-'));
                    return keys.map(k => ({
                        name: k,
                        size: localStorage.getItem(k).length,
                        modified: new Date().toISOString()
                    })).reverse();
                } catch (error) {
                    return [];
                }
            }
        }

        // ============================================
        // 2. API Client
        // ============================================

        class PuterAPIClient {
            constructor() {
                this.storage = new PuterStorageManager();
            }

            async init() {
                await this.storage.init();
            }

            async generateIsland(seed = null) {
                const islandData = {
                    id: this.generateUUID(),
                    seed: seed || Math.random(),
                    terrain: this.generateTerrain(130, 105, seed),
                    created: new Date().toISOString(),
                    buildings: []
                };
                console.log(`✓ Island generated: ${islandData.id}`);
                return islandData;
            }

            generateTerrain(width, height, seed) {
                const terrain = Array(height).fill(null).map(() => Array(width).fill(0));
                const centerX = width / 2;
                const centerY = height / 2;
                const islandRadius = Math.min(centerX, centerY) * 0.7;

                for (let y = 0; y < height; y++) {
                    for (let x = 0; x < width; x++) {
                        const dx = x - centerX;
                        const dy = y - centerY;
                        const distance = Math.sqrt(dx * dx + dy * dy);
                        
                        const noise = this.seededNoise(x, y, seed);
                        const falloff = 1 - (distance / islandRadius);
                        const value = falloff + (noise - 0.5) * 0.4;

                        if (value < 0) {
                            terrain[y][x] = 0;
                        } else if (value < 0.15) {
                            terrain[y][x] = 1;
                        } else if (value < 0.4) {
                            terrain[y][x] = 2;
                        } else {
                            terrain[y][x] = 3;
                        }
                    }
                }
                return terrain;
            }

            seededNoise(x, y, seed) {
                const n = Math.sin(x * 12.9898 + y * 78.233 + (seed || 0)) * 43758.5453;
                return n - Math.floor(n);
            }

            async placeBuilding(islandId, buildingData) {
                const building = {
                    id: this.generateUUID(),
                    islandId,
                    ...buildingData,
                    deployed: new Date().toISOString()
                };
                console.log(`✓ Building placed: ${building.id}`);
                return building;
            }

            generateUUID() {
                return 'xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx'.replace(/[xy]/g, function(c) {
                    const r = Math.random() * 16 | 0;
                    const v = c === 'x' ? r : (r & 0x3 | 0x8);
                    return v.toString(16);
                });
            }
        }

        // ============================================
        // 3. Phaser Game Scene (RTS IMPROVED)
        // ============================================

        class PuterIslandScene extends Phaser.Scene {
            constructor() {
                super('PuterIslandScene');
                this.api = new PuterAPIClient();
                this.storage = this.api.storage;
                
                this.islandData = null;
                this.islandId = null;
                this.buildings = [];
                this.selectedBuilding = null;
                this.buildingPreview = null;
                this.isLoading = false;

                this.GRID_SIZE = 32;
                this.ISLAND_WIDTH = 130;
                this.ISLAND_HEIGHT = 105;

                // RTS Camera Control
                this.cameraSpeed = 400;
                this.keys = {};
                this.isRightMouseDown = false;
                this.rightMouseStartX = 0;
                this.rightMouseStartY = 0;
                this.cameraStartX = 0;
                this.cameraStartY = 0;
            }

            async preload() {
                await this.api.init();
            }

            async create() {
                try {
                    this.isLoading = true;
                    
                    const savedState = await this.storage.loadLatestGameState();
                    
                    if (savedState && savedState.islandId) {
                        this.islandId = savedState.islandId;
                        this.islandData = savedState.islandData;
                        this.buildings = savedState.buildings || [];
                    } else {
                        this.islandData = await this.api.generateIsland();
                        this.islandId = this.islandData.id;
                    }

                    this.setupPhysics();
                    this.createTerrainDisplay();
                    this.setupCamera();
                    this.setupRTSInput();
                    this.setupMinimap();

                    // Store reference for UI updates
                    window.gameScene = this;
                    updateGameUI();

                    this.isLoading = false;
                    console.log('✓ Game scene ready with RTS controls');
                } catch (error) {
                    console.error('Scene error:', error);
                }
            }

            setupPhysics() {
                this.physics.world.setBounds(
                    0, 0,
                    this.ISLAND_WIDTH * this.GRID_SIZE,
                    this.ISLAND_HEIGHT * this.GRID_SIZE
                );
            }

            createTerrainDisplay() {
                const graphics = this.make.graphics({ x: 0, y: 0, add: false });
                const colors = [0x1a4d7a, 0xc2b280, 0x5fa354, 0x2d5016];

                for (let y = 0; y < this.ISLAND_HEIGHT; y++) {
                    for (let x = 0; x < this.ISLAND_WIDTH; x++) {
                        const terrainType = this.islandData.terrain[y][x];
                        const color = colors[terrainType] || 0xffffff;
                        
                        graphics.fillStyle(color, 1);
                        graphics.fillRect(x * this.GRID_SIZE, y * this.GRID_SIZE, this.GRID_SIZE, this.GRID_SIZE);

                        if (x >= 60 && x <= 70 && y >= 50 && y <= 60) {
                            graphics.lineStyle(1, 0x00ff00, 0.3);
                            graphics.strokeRect(x * this.GRID_SIZE, y * this.GRID_SIZE, this.GRID_SIZE, this.GRID_SIZE);
                        }
                    }
                }

                const worldWidth = this.ISLAND_WIDTH * this.GRID_SIZE;
                const worldHeight = this.ISLAND_HEIGHT * this.GRID_SIZE;
                graphics.generateTexture('islandTexture', worldWidth, worldHeight);
                graphics.destroy();

                this.add.image(0, 0, 'islandTexture').setOrigin(0).setDepth(0);

                this.buildings.forEach(b => this.renderBuilding(b));
            }

            renderBuilding(building) {
                const colors = {
                    'house': 0x0088ff,
                    'tower': 0xff3333,
                    'farm': 0xffdd00,
                    'market': 0xff8800
                };

                this.add.rectangle(
                    building.worldX,
                    building.worldY,
                    this.GRID_SIZE - 4,
                    this.GRID_SIZE - 4,
                    colors[building.type] || 0xffffff,
                    0.9
                ).setDepth(15);
            }

            setupCamera() {
                const camera = this.cameras.main;
                const centerX = (this.ISLAND_WIDTH / 2) * this.GRID_SIZE;
                const centerY = (this.ISLAND_HEIGHT / 2) * this.GRID_SIZE;
                
                camera.centerOn(centerX, centerY);
                camera.setZoom(1);
                camera.setBounds(0, 0,
                    this.ISLAND_WIDTH * this.GRID_SIZE,
                    this.ISLAND_HEIGHT * this.GRID_SIZE
                );
            }

            setupRTSInput() {
                // WASD Keys
                this.input.keyboard.on('keydown-W', () => this.keys['W'] = true);
                this.input.keyboard.on('keyup-W', () => this.keys['W'] = false);
                this.input.keyboard.on('keydown-A', () => this.keys['A'] = true);
                this.input.keyboard.on('keyup-A', () => this.keys['A'] = false);
                this.input.keyboard.on('keydown-S', () => {
                    if (event.ctrlKey || event.metaKey) return;
                    this.keys['S'] = true;
                });
                this.input.keyboard.on('keyup-S', () => this.keys['S'] = false);
                this.input.keyboard.on('keydown-D', () => this.keys['D'] = true);
                this.input.keyboard.on('keyup-D', () => this.keys['D'] = false);

                // Arrow Keys
                this.input.keyboard.on('keydown-UP', () => this.keys['UP'] = true);
                this.input.keyboard.on('keyup-UP', () => this.keys['UP'] = false);
                this.input.keyboard.on('keydown-LEFT', () => this.keys['LEFT'] = true);
                this.input.keyboard.on('keyup-LEFT', () => this.keys['LEFT'] = false);
                this.input.keyboard.on('keydown-DOWN', () => this.keys['DOWN'] = true);
                this.input.keyboard.on('keyup-DOWN', () => this.keys['DOWN'] = false);
                this.input.keyboard.on('keydown-RIGHT', () => this.keys['RIGHT'] = true);
                this.input.keyboard.on('keyup-RIGHT', () => this.keys['RIGHT'] = false);

                // Building Selection
                this.input.keyboard.on('keydown-1', () => this.selectedBuilding = 'house');
                this.input.keyboard.on('keydown-2', () => this.selectedBuilding = 'tower');
                this.input.keyboard.on('keydown-3', () => this.selectedBuilding = 'farm');
                this.input.keyboard.on('keydown-4', () => this.selectedBuilding = 'market');

                // Game Controls
                this.input.keyboard.on('keydown-R', () => this.resetCamera());
                this.input.keyboard.on('keydown-ESC', () => this.clearBuildingPreview());
                this.input.keyboard.on('keydown-I', () => this.toggleInfoPanel());
                this.input.keyboard.on('keydown-M', () => this.toggleMinimap());

                // Zoom with Scroll (FIXED - only on scroll direction)
                this.input.on('wheel', (pointer, gameObjects, deltaY) => {
                    const camera = this.cameras.main;
                    const zoomFactor = deltaY > 0 ? 0.9 : 1.1; // Zoom out or in
                    const newZoom = Phaser.Math.Clamp(
                        camera.zoom * zoomFactor,
                        0.5, 3
                    );
                    camera.setZoom(newZoom);
                }, this);

                // Right-Click Drag for camera pan (RTS style)
                this.input.on('pointerdown', (pointer) => {
                    if (pointer.rightButtonDown()) {
                        this.isRightMouseDown = true;
                        this.rightMouseStartX = pointer.x;
                        this.rightMouseStartY = pointer.y;
                        const camera = this.cameras.main;
                        this.cameraStartX = camera.scrollX;
                        this.cameraStartY = camera.scrollY;
                    }
                });

                this.input.on('pointerup', (pointer) => {
                    if (!pointer.rightButtonDown()) {
                        this.isRightMouseDown = false;
                    }
                });

                this.input.on('pointermove', (pointer) => {
                    // Right-click drag pan
                    if (this.isRightMouseDown) {
                        const camera = this.cameras.main;
                        const dx = (this.rightMouseStartX - pointer.x) / camera.zoom;
                        const dy = (this.rightMouseStartY - pointer.y) / camera.zoom;
                        camera.scrollX = this.cameraStartX + dx;
                        camera.scrollY = this.cameraStartY + dy;
                    }

                    // Building preview
                    this.updateBuildingPreview(pointer);
                });

                // Left-click placement
                this.input.on('pointerdown', (pointer) => {
                    if (pointer.leftButtonDown() && !this.isLoading) {
                        this.placeBuildingAtPointer(pointer);
                    }
                });

                // Save game on S key (with guard for text inputs)
                document.addEventListener('keydown', (e) => {
                    if (e.key === 's' || e.key === 'S') {
                        if (!(e.target instanceof HTMLInputElement || e.target instanceof HTMLTextAreaElement)) {
                            if (!e.ctrlKey && !e.metaKey) {
                                this.saveGame();
                                e.preventDefault();
                            }
                        }
                    }
                });
            }

            // Update camera position from keyboard input in update loop
            update() {
                const camera = this.cameras.main;
                const dt = this.sys.game.loop.delta / 1000; // Delta time in seconds
                const moveDistance = this.cameraSpeed * dt;

                // Move based on WASD or Arrow keys
                if (this.keys['W'] || this.keys['UP']) {
                    camera.scrollY -= moveDistance;
                }
                if (this.keys['S'] || this.keys['DOWN']) {
                    camera.scrollY += moveDistance;
                }
                if (this.keys['A'] || this.keys['LEFT']) {
                    camera.scrollX -= moveDistance;
                }
                if (this.keys['D'] || this.keys['RIGHT']) {
                    camera.scrollX += moveDistance;
                }

                // Clamp camera to bounds
                const bounds = this.physics.world.bounds;
                const width = this.cameras.main.displayWidth / camera.zoom;
                const height = this.cameras.main.displayHeight / camera.zoom;
                
                camera.scrollX = Phaser.Math.Clamp(
                    camera.scrollX,
                    0,
                    Math.max(0, bounds.width - width)
                );
                camera.scrollY = Phaser.Math.Clamp(
                    camera.scrollY,
                    0,
                    Math.max(0, bounds.height - height)
                );
            }

            updateBuildingPreview(pointer) {
                if (!this.selectedBuilding) return;

                const camera = this.cameras.main;
                const worldX = (pointer.x / camera.zoom) + camera.scrollX;
                const worldY = (pointer.y / camera.zoom) + camera.scrollY;
                const snappedX = Math.round(worldX / this.GRID_SIZE) * this.GRID_SIZE - this.GRID_SIZE / 2;
                const snappedY = Math.round(worldY / this.GRID_SIZE) * this.GRID_SIZE - this.GRID_SIZE / 2;

                if (!this.buildingPreview) {
                    this.buildingPreview = this.add.rectangle(
                        snappedX, snappedY,
                        this.GRID_SIZE - 2,
                        this.GRID_SIZE - 2,
                        0xffff00, 0.4
                    ).setDepth(50);
                }

                this.buildingPreview.setPosition(snappedX, snappedY);

                const gridX = Math.floor(snappedX / this.GRID_SIZE);
                const gridY = Math.floor(snappedY / this.GRID_SIZE);
                const isValid = this.isValidPlacement(gridX, gridY);

                this.buildingPreview.setFillStyle(isValid ? 0x00ff00 : 0xff0000, 0.4);
            }

            async placeBuildingAtPointer(pointer) {
                if (!this.selectedBuilding) return;

                const camera = this.cameras.main;
                const worldX = (pointer.x / camera.zoom) + camera.scrollX;
                const worldY = (pointer.y / camera.zoom) + camera.scrollY;
                const snappedX = Math.round(worldX / this.GRID_SIZE) * this.GRID_SIZE - this.GRID_SIZE / 2;
                const snappedY = Math.round(worldY / this.GRID_SIZE) * this.GRID_SIZE - this.GRID_SIZE / 2;

                const gridX = Math.floor(snappedX / this.GRID_SIZE);
                const gridY = Math.floor(snappedY / this.GRID_SIZE);

                if (!this.isValidPlacement(gridX, gridY)) {
                    showNotification('Cannot place building here!');
                    return;
                }

                const building = await this.api.placeBuilding(this.islandId, {
                    type: this.selectedBuilding,
                    gridX,
                    gridY,
                    worldX: snappedX + this.GRID_SIZE / 2,
                    worldY: snappedY + this.GRID_SIZE / 2
                });

                this.buildings.push(building);
                this.renderBuilding(building);
                showNotification(`${building.type.toUpperCase()} placed! (${gridX}, ${gridY})`);
            }

            isValidPlacement(gridX, gridY) {
                if (gridX < 60 || gridX > 70 || gridY < 50 || gridY > 60) {
                    return false;
                }

                for (let building of this.buildings) {
                    const dist = Math.hypot(gridX - building.gridX, gridY - building.gridY);
                    if (dist < 1.5) return false;
                }

                return true;
            }

            async saveGame() {
                const gameState = {
                    islandId: this.islandId,
                    islandData: this.islandData,
                    buildings: this.buildings,
                    timestamp: new Date().toISOString()
                };

                const result = await this.storage.saveGameState(gameState);
                if (result.success) {
                    showNotification('✓ Game Saved!');
                }
            }

            clearBuildingPreview() {
                if (this.buildingPreview) {
                    this.buildingPreview.destroy();
                    this.buildingPreview = null;
                }
                this.selectedBuilding = null;
                showNotification('Building deselected');
            }

            resetCamera() {
                const centerX = (this.ISLAND_WIDTH / 2) * this.GRID_SIZE;
                const centerY = (this.ISLAND_HEIGHT / 2) * this.GRID_SIZE;
                this.cameras.main.centerOn(centerX, centerY);
                this.cameras.main.setZoom(1);
            }

            toggleInfoPanel() {
                const panel = document.getElementById('game-info');
                panel.classList.toggle('visible');
            }

            toggleMinimap() {
                const minimap = document.getElementById('minimap');
                minimap.style.display = minimap.style.display === 'none' ? 'block' : 'none';
            }

            setupMinimap() {
                const canvas = document.getElementById('minimap-canvas');
                const ctx = canvas.getContext('2d');
                
                // Set canvas size
                canvas.width = 150;
                canvas.height = 150;

                // Render minimap
                const scale = 150 / Math.max(this.ISLAND_WIDTH, this.ISLAND_HEIGHT);
                const colors = ['#1a4d7a', '#c2b280', '#5fa354', '#2d5016'];

                for (let y = 0; y < this.ISLAND_HEIGHT; y++) {
                    for (let x = 0; x < this.ISLAND_WIDTH; x++) {
                        const terrainType = this.islandData.terrain[y][x];
                        ctx.fillStyle = colors[terrainType] || '#ffffff';
                        ctx.fillRect(x * scale, y * scale, scale, scale);
                    }
                }

                // Draw buildings on minimap
                this.buildings.forEach(b => {
                    ctx.fillStyle = '#ff0000';
                    ctx.fillRect(b.gridX * scale, b.gridY * scale, scale * 1.5, scale * 1.5);
                });
            }
        }

            document.body.appendChild(notif);

        }

        function updateGameUI() {
            setInterval(() => {
                if (window.gameScene && window.gameScene.islandData) {
                    const gs = window.gameScene;
                    const camera = gs.cameras.main;
                    
                    document.getElementById('info-island').textContent = 
                        gs.islandId.substring(0, 12) + '...';
                    document.getElementById('info-seed').textContent = 
                        gs.islandData.seed.toFixed(6);
                    document.getElementById('info-zoom').textContent = 
                        camera.zoom.toFixed(2) + 'x';
                    document.getElementById('info-buildings').textContent = 
                        gs.buildings.length + '/100';
                    document.getElementById('info-selected').textContent = 
                        gs.selectedBuilding || 'None';
                    document.getElementById('info-pos').textContent =
                        Math.round(camera.scrollX / 32) + ', ' + Math.round(camera.scrollY / 32);
                }
            }, 100);
        }

        let frameCount = 0;
        let lastTime = Date.now();

        function updateFPS() {
            frameCount++;
            const now = Date.now();
            if (now - lastTime >= 1000) {
                document.getElementById('fps-counter').textContent = 'FPS: ' + frameCount;
                frameCount = 0;
                lastTime = now;
            }
            requestAnimationFrame(updateFPS);
        }
